/**
 * Simple shuffle algorithm for arrays
 * @param items Array
 */
var shuffle =  function(items) {
  var seed, x, i;
  for (i = items.length; i > 0; i--) {
    seed = parseInt(Math.random() * i);
    // swap items
    x = items[i], items[i] = items[seed], items[seed] = x;
  }
  return items;
};

      stock = subject.current_stock
      subject.use.must_equal true
      subject.current_stock.must_equal (stock - 1)
    end
  end
end
